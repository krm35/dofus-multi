hasCooldown(args) → {boolean|undefined}

Check if a spell is available (not on cooldown) for casting
Parameters:
Name Type Description
args Object The function arguments
Properties
Name Type Description
id number The spell ID to check cooldown for
delay number The cooldown duration in turns for this spell
Throws:
Throws error if spell ID doesn't exist or delay is not provided
Type
Error
Returns:
True if spell is available, false if on cooldown, undefined if not in combat
Type
boolean | undefined
Examples
// Check if a spell with 3-turn cooldown is available
if (hasCooldown({id: 123, delay: 3})) {
    await spell({id: 123, cell: enemyCell});
} else {
    console.log("Spell 123 is still on cooldown");
}
// Plan combat sequence with cooldown management
const spellRotation = [
    {id: 123, delay: 4}, // Powerful spell, 4-turn cooldown
    {id: 456, delay: 2}, // Medium spell, 2-turn cooldown
    {id: 789, delay: 1}  // Basic spell, 1-turn cooldown
];

for (const spellData of spellRotation) {
    if (hasCooldown(spellData) && getActionPoints() >= getSpellCost(spellData.id)) {
        await spell({id: spellData.id, cell: enemyCell});
        break; // Cast first available spell
    }
}
// Check multiple spells and choose the best available
const availableSpells = [
    {id: 111, delay: 5, damage: 'high'},
    {id: 222, delay: 3, damage: 'medium'},
    {id: 333, delay: 1, damage: 'low'}
].filter(spell => hasCooldown(spell));

if (availableSpells.length > 0) {
    const bestSpell = availableSpells[0]; // Use highest damage available
    await spell({id: bestSpell.id, cell: enemyCell});
}
// Safe cooldown check with error handling
try {
    if (hasCooldown({id: 456, delay: 3})) {
        await spell({id: 456, cell: targetCell});
    } else {
        // Use alternative spell
        await spell({id: 789, cell: targetCell});
    }
} catch (error) {
    console.error("Spell check failed:", error.message);
}
// Conditional casting based on combat state
if (isInFight() && c.turn > 0) {
    const canCastHeal = hasCooldown({id: healSpellId, delay: 4});
    const canCastAttack = hasCooldown({id: attackSpellId, delay: 2});

    if (myHealth < 50 && canCastHeal) {
        await spell({id: healSpellId, cell: myCell()});
    } else if (canCastAttack) {
        await spell({id: attackSpellId, cell: enemyCell});
    }
}
// Track cooldowns for strategic planning
const importantSpells = [
    {id: 100, delay: 6, name: "Ultimate Attack"},
    {id: 200, delay: 4, name: "Shield"},
    {id: 300, delay: 3, name: "Heal"}
];

importantSpells.forEach(spell => {
    const available = hasCooldown(spell);
    console.log(`${spell.name}: ${available ? 'Available' : 'On Cooldown'}`);
});
